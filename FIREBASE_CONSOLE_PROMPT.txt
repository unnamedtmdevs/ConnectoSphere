  match /databases/{database}/documents {
    
    // Users - —á–∏—Ç–∞—Ç—å –º–æ–≥—É—Ç –≤—Å–µ, –ø–∏—Å–∞—Ç—å —Ç–æ–ª—å–∫–æ —Å–≤–æ–π –ø—Ä–æ—Ñ–∏–ª—å

    // Circles - —á–∏—Ç–∞—Ç—å –≤—Å–µ, —Å–æ–∑–¥–∞–≤–∞—Ç—å –∞–≤—Ç–æ—Ä–∏–∑–æ–≤–∞–Ω–Ω—ã–µ, –æ–±–Ω–æ–≤–ª—è—Ç—å –∞–≤—Ç–æ—Ä–∏–∑–æ–≤–∞–Ω–Ω—ã–µ

      allow update: if request.auth != null;
      allow delete: if false; // –ó–∞–ø—Ä–µ—â–∞–µ–º —É–¥–∞–ª–µ–Ω–∏–µ
    }
    
    // Posts - —á–∏—Ç–∞—Ç—å –≤—Å–µ, —Å–æ–∑–¥–∞–≤–∞—Ç—å –∞–≤—Ç–æ—Ä–∏–∑–æ–≤–∞–Ω–Ω—ã–µ, —É–¥–∞–ª—è—Ç—å —Ç–æ–ª—å–∫–æ –∞–≤—Ç–æ—Ä

      allow read: if true;
      allow create: if request.auth != null;
      allow update: if request.auth != null;
      allow delete: if request.auth != null && 
                       resource.data.authorID == request.auth.uid;
    }
    
    // Comments - —á–∏—Ç–∞—Ç—å –≤—Å–µ, —Å–æ–∑–¥–∞–≤–∞—Ç—å –∞–≤—Ç–æ—Ä–∏–∑–æ–≤–∞–Ω–Ω—ã–µ, –∏–∑–º–µ–Ω—è—Ç—å/—É–¥–∞–ª—è—Ç—å —Ç–æ–ª—å–∫–æ –∞–≤—Ç–æ—Ä
    match /comments/{commentId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update: if request.auth != null && 
                       resource.data.authorID == request.auth.uid;
      allow delete: if request.auth != null && 
                       resource.data.authorID == request.auth.uid;
    }
  }
}
